            this.filas = [];
            this.campos = [];

    methods: {
        ordena(campo) {
            this.filas = this.filas.sort((a, b) => {

borrar(id) {
    var url = window.location.href;
    var self = this;
    axios.delete(url + "/" + id)
        .then(function (response) {
            self.filas = response.data;
    });
